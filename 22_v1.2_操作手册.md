## 一、环境搭建

### 1.1 克隆项目

```bash
git clone https://github.com/biandeshen/Plector.git
cd Plector
```

### 1.2 创建虚拟环境

```bash
python -m venv venv

# Windows
venv\Scripts\activate

# Linux / macOS
source venv/bin/activate
```

### 1.3 安装依赖

```bash
pip install -r requirements.txt
```

### 1.4 配置 LLM

编辑 `config/config.yaml`：

```yaml
llm:
  provider: "${LLM_PROVIDER:-ollama}"
  max_iterations: 10

  ollama:
    base_url: "http://localhost:11434"
    model: "${OLLAMA_MODEL:-qwen3:4b}"

  openai:
    api_key: "${OPENAI_API_KEY}"
    model: "${OPENAI_MODEL:-gpt-4o-mini}"
    base_url: "${OPENAI_BASE_URL:-https://api.openai.com/v1}"

  anthropic:
    api_key: "${ANTHROPIC_API_KEY}"
    model: "${ANTHROPIC_MODEL:-claude-sonnet-4-20250514}"
```

设置环境变量（三选一）：

```bash
# 方案 1：Ollama（本地，免费）
ollama pull qwen3:4b
ollama serve

# 方案 2：OpenAI
export OPENAI_API_KEY="sk-xxx"
export LLM_PROVIDER="openai"

# 方案 3：Anthropic
export ANTHROPIC_API_KEY="sk-ant-xxx"
export LLM_PROVIDER="anthropic"
```

        "type": "object",
        "properties": {

        "additionalProperties": false

```

### 3.3 implementation.py 模板

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
技能描述

Author: Plector
Version: 1.0.0
Created: 2026-04-04
"""

import logging
from typing import Any, Dict

from core.event_bus import get_event_bus

logger = logging.getLogger(__name__)


class SkillHandler:
    """技能处理器"""

    def __init__(self):
        self.name = "my_skill"

    async def my_tool(self, param1: str) -> Dict[str, Any]:
        """
        工具描述

        参数:
            param1: 参数描述

        返回:
            {"success": bool, "data": {...}, "error": str or None}
        """
        try:
            # 业务逻辑
            result = f"处理结果: {param1}"

            # 发布事件
            bus = get_event_bus()
            await bus.publish("my_skill.completed", {
                "param1": param1,
                "result": result,
            }, source="my_skill")

            return {
                "success": True,
                "data": {"result": result},
                "error": None,
            }
        except Exception as e:
            logger.error(f"工具执行失败: {e}", exc_info=True)
            return {"success": False, "data": None, "error": str(e)}
```

### 4.2 MCP Server 模板

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
MCP Server 模板

协议：JSON-RPC 2.0 over stdio
"""

import json
import sys


# 工具定义
TOOLS = [
    {
        "name": "my_tool",
        "description": "工具描述",
        "inputSchema": {
            "type": "object",
            "properties": {
                "param1": {"type": "string", "description": "参数描述"},
            },
            "required": ["param1"],
            "additionalProperties": False,
        },
    },
]


def handle_my_tool(args: dict) -> str:
    """工具实现"""
    param1 = args.get("param1", "")
    return f"处理结果: {param1}"


TOOL_HANDLERS = {
    "my_tool": handle_my_tool,
}


def handle_initialize(req_id) -> dict:
    return {
        "jsonrpc": "2.0",
        "id": req_id,
        "result": {
            "protocolVersion": "2024-11-05",
            "capabilities": {"tools": {"listChanged": False}},
            "serverInfo": {"name": "my-server", "version": "1.0.0"},
        },
    }


def handle_tools_list(req_id) -> dict:
    return {
        "jsonrpc": "2.0",
        "id": req_id,
        "result": {"tools": TOOLS},
    }


def handle_tools_call(req_id, params: dict) -> dict:
    tool_name = params.get("name", "")
    arguments = params.get("arguments", {})
    handler = TOOL_HANDLERS.get(tool_name)
    if not handler:
        return {
            "jsonrpc": "2.0",
            "id": req_id,
            "error": {"code": -32601, "message": f"未知工具: {tool_name}"},
        }
    try:
        result_text = handler(arguments)
        return {
            "jsonrpc": "2.0",
            "id": req_id,
            "result": {"content": [{"type": "text", "text": result_text}]},
        }
    except Exception as e:
        return {
            "jsonrpc": "2.0",
            "id": req_id,
            "result": {"content": [{"type": "text", "text": f"错误: {e}"}]},
        }


def handle_request(request: dict) -> dict:
    """处理 JSON-RPC 2.0 请求"""
    method = request.get("method", "")
    params = request.get("params", {})
    req_id = request.get("id")

    if method == "initialize":
        return handle_initialize(req_id)
    elif method == "notifications/initialized":
        return None
    elif method == "tools/list":
        return handle_tools_list(req_id)
    elif method == "tools/call":
        return handle_tools_call(req_id, params)
    else:
        return {
            "jsonrpc": "2.0",
            "id": req_id,
            "error": {"code": -32601, "message": f"未知方法: {method}"},
        }


def main():
    """主循环"""
    while True:
        try:
            line = sys.stdin.readline()
            if not line:
                break
            line = line.strip()
            if not line:
                continue
            request = json.loads(line)
            response = handle_request(request)
            if response is not None:
                sys.stdout.write(json.dumps(response) + "\n")
                sys.stdout.flush()
        except json.JSONDecodeError:
            sys.stdout.write(json.dumps({
                "jsonrpc": "2.0", "id": None,
                "error": {"code": -32700, "message": "JSON 解析失败"},
            }) + "\n")
            sys.stdout.flush()
        except Exception:
            break


if __name__ == "__main__":
    main()
```

## 六、Git 工作流

### 6.1 提交规范

```
<type>(<scope>): <subject>
```

**type**：
- `feat`：新功能
- `fix`：修复
- `docs`：文档
- `refactor`：重构
- `test`：测试
- `chore`：杂项

**示例**：
```
feat(my_skill): 添加 my_skill 技能（1 个工具）
fix(security): 修复 check_safe_path 绕过漏洞
docs: 更新 Design v1.2
refactor(servers): 改用纯 Python 实现 MCP 协议
```

### 6.2 提交流程

```bash
# 1. 创建分支（可选）
git checkout -b feat/my-skill

git add -A

### 9.4 闭环配置

```yaml
# config/closed_loops.yaml

error_record_loop:
  trigger_on: ["test.failed", "skill.failed"]
  entry: "store_error"
  max_iterations: 3
  nodes:
    store_error:
      type: "skill"
      skill: "error_knowledge"
      method: "store_error"
      next: "classify_error"
    classify_error:
      type: "skill"
      skill: "error_knowledge"
      method: "classify_error"
      next: "end"
    end:
      type: "end"

health_check_loop:
  trigger_on: ["health.degraded"]
  entry: "check_health"
  max_iterations: 3
  nodes:
    check_health:
      type: "skill"

      method: "check_health"
      next: "end"

      type: "end"
