## 第一步：创建记忆衰减模块

创建 `core/memory_decay.py`：

```python
#!/usr/bin/env python3
"""
记忆衰减模块 - 基于对话轮次的艾宾浩斯遗忘曲线

核心逻辑：
    对 AI 来说"时间"= 对话轮次，不是日历时间
    记忆强度随距离当前轮次衰减
    每次访问（复习）会加强记忆

Author: Plector
Version: 2.0.0
"""

import math
import logging
from typing import Any

logger = logging.getLogger(__name__)

# 强度等级阈值
STRENGTH_ALIVE = 0.8
STRENGTH_NORMAL = 0.5
STRENGTH_FADING = 0.2
STRENGTH_FORGOTTEN = 0.05

# 默认稳定性系数（轮次）
DEFAULT_STABILITY = 10.0


def calculate_strength(
    current_turn: int,
    memory_turn: int,
    access_count: int = 1,
    stability: float = DEFAULT_STABILITY,
) -> float:
    """基于对话轮次计算记忆强度"""
    turns_ago = max(0, current_turn - memory_turn)
    adjusted_stability = stability * (1 + math.log1p(access_count))
    strength = math.exp(-turns_ago / adjusted_stability)
    return max(0.0, min(1.0, strength))


def get_strength_level(strength: float) -> str:
    """获取强度等级"""
    if strength >= STRENGTH_ALIVE:
        return "alive"
    elif strength >= STRENGTH_NORMAL:
        return "normal"
    elif strength >= STRENGTH_FADING:
        return "fading"
    else:
        return "forgotten"


def build_memory_metadata(
    memory_type: str = "knowledge",
    current_turn: int = 0,
    stability: float = DEFAULT_STABILITY,
) -> dict[str, Any]:
    """构建记忆元数据"""
    return {
        "created_at_turn": current_turn,
        "last_accessed_turn": current_turn,
        "access_count": 1,
        "stability": stability,
        "strength": 1.0,
        "strength_level": "alive",
        "memory_type": memory_type,
        "associations": "",
        "scenarios": "",
        "opposites": "",
        "reinterpreted": False,
    }


def update_on_access(metadata: dict[str, Any], current_turn: int) -> dict[str, Any]:
    """访问时更新记忆元数据"""
    metadata["last_accessed_turn"] = current_turn
    metadata["access_count"] = metadata.get("access_count", 0) + 1
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=current_turn,
        access_count=metadata["access_count"],
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)
    return metadata


def decay_memory(metadata:[str, Any]:
    """计算记忆当前衰减后的强度"""
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)
    return metadata


def should_cleanup(
    metadata: dict[str, Any],
    current_turn: int,
    min_strength: float = STRENGTH_FORGOTTEN,
) -> bool:
    """判断记忆是否应该被清理"""
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    return strength < min_strength
```

#!/usr/bin/env python3
"""
记忆联想模块 - 6 种联想模式

联想模式：
    1. 语义相似 - ChromaDB 向量检索
    2. 触景生情 - 用当前上下文关键词触发
    3. 触类旁通 - 用关联概念扩展检索
    4. 时序联想 - 同一轮次附近的记忆
    5. dict[str, Any], current_turn: int) -> dict 温故知新 - 定期重访旧记忆 + LLM 重新理解
    6. 举一反三 - 存储时 LLM 生成适用场景
    7. 推陈出新 - LLM 融合多条记忆生成新方案
    8. 恍然大悟 - 新旧记忆矛盾时重新标记

Author: Plector
Version: 2.0.0
"""

import json
import logging
from typing import Any

logger = logging.getLogger(__name__)


# ============================================================
# LLM 调用辅助
# ============================================================

async def _call_llm(prompt: str, llm_client) -> str:
    """调用 LLM 生成文本"""
    try:
        result = await llm_client.chat(
            messages=[{"role": "user", "content": prompt}],
            max_tokens=500,
        )
        if isinstance(result, dict):
            return result.get("content", "")
        return str(result)
    except Exception as e:
        logger.error(f"LLM 调用失败: {e}", exc_info=True)
        return ""


async def _call_llm_json(prompt: str, llm_client) -> dict:
    """调用 LLM 生成 JSON"""
    text = await _call_llm(prompt, llm_client)
    if not text:
        return {}
    try:
        # 尝试提取 JSON
        text = text.strip()
        if text.startswith("```"):
            text = text.split("\n", 1)[1]
            text = text.rsplit("```", 1)[0]
        return json.loads(text)
    except json.JSONDecodeError:
        logger.warning(f"LLM 返回非 JSON: {text[:100]}")
        return {}

def search_by_triggers(
    collection,
    query_keywords: list[str],
    current_turn: int,
    n_results: int = 5,
) -> list[dict]:
    """触景生情：用关键词触发相关记忆"""
    if not query_keywords:
        return []

    results = []
    seen_ids = set()

    for keyword in query_keywords[:5]:
        try:
            hits = collection.query(query_texts=[keyword], n_results=3)
            if hits["ids"] and hits["ids"][0]:
                for i, mid in enumerate(hits["ids"][0]):
                    if mid not in seen_ids:
                        seen_ids.add(mid)
                        results.append({
                            "id": mid,
                            "content": hits["documents"][0][i],
                            "metadata": hits["metadatas"][0][i] if hits["metadatas"] else {},
                            "source": "trigger",
                            "trigger_keyword": keyword,
                        })
        except Exception:
            continue

    return results[:n_results]


# ============================================================
# 触类旁通
# ============================================================

def search_by_associations(
    collection,
    query: str,
    current_turn: int,
    n_results: int = 5,
) -> list[dict]:
    """触类旁通：用关联概念扩展检索"""
    initial = collection.query(query_texts=[query], n_results=3)
    if not initial["ids"] or not initial["ids"][0]:
        return []

    association_keywords = []
    for metadata in (initial["metadatas"][0] if initial["metadatas"] else []):
        assoc = metadata.get("associations", "")
        if assoc:
            association_keywords.extend([a.strip() for a in assoc.split(",") if a.strip()])

    if not association_keywords:
        return []

    results = []
    seen_ids = set(initial["ids"][0])

    for keyword in association_keywords[:5]:
        try:
            hits = collection.query(query_texts=[keyword], n_results=2)
            if hits["ids"] and hits["ids"][0]:
                for i, mid in enumerate(hits["ids"][0]):
                    if mid not in seen_ids:
                        seen_ids.add(mid)
                        results.append({
                            "id": mid,
                            "content": hits["documents"][0][i],
                            "metadata": hits["metadatas"][0][i] if hits["metadatas"] else {},
                            "source": "association",
                            "via_keyword": keyword,
                        })
        except Exception:
            continue

    return results[:n_results]

def search_by_temporal(
    collection,
    current_turn: int: int = 5,
) -> list[dict]:
    """时序联想：检索当前轮次附近的记忆"""
    try:
        all_items =,
    window: int = 5,
    n_results collection.get()
    except Exception:
        return []

    if not all_items["ids"]:
        return []

    results = []
    for i, mid in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        created_turn = metadata.get("created_at_turn", 0)

        if abs(current_turn - created_turn) <= window:
            results.append({
                "id": mid,
                "content": all_items["documents"][i],
                "metadata": metadata,
                "source": "temporal",
                "turns_ago": current_turn - created_turn,
            })

    results.sort(key=lambda x: abs(x["turns_ago"]))
    return results[:n_results]

def search_by_review(
    collection,
    current_turn: int,
    review_interval: int = 50,
    n_results: int = 5,
) -> list[dict]:
    """温故知新：定期重访旧记忆"""
    if current_turn % review_interval != 0:
        return []

    try:
        all_items = collection.get()
    except Exception:
        return []

    if not all_items["ids"]:
        return []

    results = []
    for i, mid in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        last_accessed = metadata.get("last_accessed_turn", 0)

        results.append({
            "id": mid,
            "content": all_items["documents"][i],
            "metadata": metadata,
            "source": "review",
            "turns_since_access": current_turn - last_accessed,
        })

    results.sort(key=lambda x: x["turns_since_access"], reverse=True)
    return results[:n_results]


async def reinterpret_memory(
    memory: dict,
    current_context: str,
    llm_client,
) -> dict:
    """
    温故知新：用 LLM 重新理解旧记忆

    给 LLM 旧记忆 + 当前上下文，让它发现新的关联
    """
    prompt = f"""分析以下旧记忆和当前上下文，发现新的关联。

旧记忆内容：
{memory.get("content", "")}

当前上下文：
{current_context}

请以 JSON 格式返回：
{{
  "new_associations": "新发现的关联概念，逗号分隔",
  "new_scenarios": "新发现的适用场景，逗号分隔",
  "insight": "新的理解或洞察，一句话",
  "has_changed": true/false
}}

只返回 JSON，不要其他内容。"""

    result = await _call_llm_json(prompt, llm_client)

    if result and result.get("has_changed"):
        metadata = memory.get("metadata", {})
        existing_assoc = metadata.get("associations", "")
        new_assoc = result.get("new_associations", "")
        if new_assoc:
            merged = list(set(
                [a.strip() for a in (existing_assoc + "," + new_assoc).split(",") if a.strip()]
            ))
            metadata["associations"] = ",".join(merged)

        existing_scenarios = metadata.get("scenarios", "")
        new_scenarios = result.get("new_scenarios", "")
        if new_scenarios:
            merged = list(set(
                [s.strip() for s in (existing_scenarios + "," + new_scenarios).split(",") if s.strip()]
            ))
            metadata["scenarios"] = ",".join(merged)

        metadata["reinterpreted"] = True
        metadata["last_interpretation"] = result.get("insight", "")
        memory["metadata"] = metadata

    return memory

async def detect_conflicts(
    new_content: str,
    existing_memories: list[dict],
    llm_client,
) -> list[dict]:
    """
    恍然大悟：检测新内容与已有记忆的矛盾
    """
    if not existing_memories:
        return []

    memory_texts = []
    for i, m in enumerate(existing_memories):
        memory_texts.append(f"[记忆{i+1}] {m.get('content', '')}")

    prompt = f"""检测以下新内容是否与已有记忆矛盾。

新内容：
{new_content}

已有记忆：
{chr(10).join(memory_texts)}

请以 JSON 格式返回：
{{
  "has_conflicts": true/false,
  "conflicts": [
    {{
      "memory_index": 1,
      "conflict_description": "矛盾描述",
      "resolution": "建议如何解决"
    }}
  ]
}}

只返回 JSON，不要其他内容。如果没有矛盾，返回 {{"has_conflicts": false, "conflicts": []}}。"""

    result = await _call_llm_json(prompt, llm_client)

    if result and result.get("has_conflicts"):
        conflicts = []
        for c in result.get("conflicts", []):
            idx = c.get("memory_index", 1) - 1
            if 0 <= idx < len(existing_memories):
                conflicts.append({
                    "memory_id": existing_memories[idx].get("id"),
                    "memory_content": existing_memories[idx].get("content", "")[:100],
                    "description": c.get("conflict_description", ""),
                    "resolution": c.get("resolution", ""),
                })
        return conflicts

    return []

# 多路召回合并
# ============================================================

def merge_and_rank(
    *result_lists: list[dict],
    current_turn: int = 0,
    n_results: int = 10,
) -> list[dict]:
    """合并多路召回结果，去重并按综合分数排序"""
    source_weights = {
        "semantic": 0.4,
        "trigger": 0.3,
        "association": 0.2,
        "temporal": 0.15,
        "review": 0.1,
    }

    merged = {}
    for results in result_lists:
        for item in results:
            mid = item["id"]
            if mid not in merged:
                merged[mid] = item
                merged[mid]["recall_sources"] = [item.get("source", "semantic")]
            else:
                if item.get("source") not in merged[mid]["recall_sources"]:
                    merged[mid]["recall_sources"].append(item.get("source", "semantic"))

    for mid, item in merged.items():
        sources = item["recall_sources"]
        source_score = sum(source_weights.get(s, 0.1) for s in sources)
        strength = item.get("metadata", {}).get("strength", 1.0)
        item["combined_score"] = round(source_score * (0.3 + 0.7 * strength), 3)

    ranked = sorted(merged.values(), key=lambda x: x["combined_score"], reverse=True)
    return ranked[:n_results]

```python
    def _semantic_search(self, query, where, current_turn, n_results):
        """语义相似检索"""
        results = self.collection.query(
            query_texts=[query], n_results=n_results, where=where if where else None,
        )

        if not results["ids"] or not results["ids"][0]:
            return []

        memories = []
        for i in range(len(results["ids"][0])):
            metadata = results["metadatas"][0][i] if results["metadatas"] else {}

            if "last_accessed_turn" in metadata:
                metadata = decay_memory(metadata, current_turn)
                strength = metadata.get("strength", 1.0)
            else:
                strength = 1.0

            distance = results["distances"][0][i] if results["distances"] else 1.0
            similarity = max(0, 1 - distance)

            memories.append({
                "id": results["ids"][0][i],
                "content": results["documents"][0][i],
                "metadata": metadata,
                "similarity": round(similarity, 3),
                "strength": round(strength, 3),
                "strength_level": get_strength_level(strength),
                "source": "semantic",
                "combined_score": round(similarity * (0.3 + 0.7 * strength), 3),
            })

        return memories

    def _filter_by_strength(self, memories, min_strength):
        """按强度过滤"""
        filtered = [m for m in memories if m.get("strength", 1.0) >= min_strength]
        filtered.sort(key=lambda x: x.get("combined_score", 0), reverse=True)
        return filtered

    @staticmethod
    def _extract_keywords(query: str) -> list[str]:
        """从查询中提取关键词"""
        import re
        words = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+', query)
        return [w for w in words if len(w) >= 2][:5]
```

```python
async def search_with_llm(
    self,
    query: str,
    task: str = "",
    n_results: int = 5,
    current_turn: int = 0,
    min_strength: float = 0.1,
    llm_client=None,
) -> dict:
    """
    搜索记忆 + LLM 推陈出新

    1. 多路召回相关记忆
    2. 如果有 task，调 LLM 融合生成新方案
    3. 温故知新时调 LLM 重新理解旧记忆
    """
    # 多路召回
    memories = self.search(
        query=query, n_results=n_results,
        current_turn=current_turn, min_strength=min_strength,
    )

    # 温故知新：重新理解旧记忆
    reinterpreted = []
    if llm_client:
        for m in memories:
            if not m.get("metadata", {}).get("reinterpreted"):
                updated = await reinterpret_memory(m, query, llm_client)
                if updated.get("metadata", {}).get("reinterpreted"):
                    reinterpreted.append(updated)
                    # 更新存储
                    self.collection.update(
                        ids=[m["id"]],
                        metadatas=[updated["metadata"]],
                    )

    # 推陈出新：融合生成新方案
    new_solution = None
    if llm_client and task and memories:
        new_solution = await synthesize_new(memories, task, llm_client)

    return {
        "memories": memories,
        "reinterpreted": reinterpreted,
        "new_solution": new_solution,
    }
```
