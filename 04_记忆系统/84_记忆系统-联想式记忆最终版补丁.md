### 旧记忆兼容的代码

在 `_decay_memory` 方法开头加：

```python
def _decay_memory(self, metadata: dict, current_turn: int) -> dict:
    """计算当前衰减强度"""
    # 兼容旧记忆（没有衰减字段）
    if "last_accessed_turn" not in metadata:
        metadata["strength"] = 1.0
        metadata["strength_level"] = "alive"
        return metadata

    strength = self._calculate_strength(
        current_turn,
        metadata.get("last_accessed_turn", 0),
        metadata.get("access_count", 1),
        metadata.get("stability", self.DEFAULT_STABILITY),
    )
    metadata["strength"] = strength
    metadata["strength_level"] = self._get_strength_level(strength)
    return metadata
```

    async def get_conversation_history(self, n_results: int = 10,
                                       current_turn: int = 0, **kwargs) -> dict:
        """获取对话历史"""
        _ = kwargs
        try:
            vm = VectorMemory()
            results = vm.search(
                query="",

                memory_type="conversation",
                current_turn=current_turn,

                use_association=False,

        except Exception as e:

    async def save_preference(self, key: str, value: str,
                              current_turn: int = 0, **kwargs) -> dict:

        try:
            vm = VectorMemory()

            return {"success": False, "data": None, "error": str(e)}

                use_association=False,
