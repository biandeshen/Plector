## 第一步：创建记忆衰减模块

创建 `core/memory_decay.py`：

```python
#!/usr/bin/env python3
"""
记忆衰减模块 - 基于对话轮次的艾宾浩斯遗忘曲线

核心逻辑：
    对 AI 来说"时间"= 对话轮次，不是日历时间
    记忆强度随距离当前轮次衰减
    每次访问（复习）会加强记忆

Author: Plector
Version: 1.0.0
"""

import math
import logging
from typing import Any

logger = logging.getLogger(__name__)

# 强度等级阈值
STRENGTH_ALIVE = 0.8      # 鲜活
STRENGTH_NORMAL = 0.5     # 正常
STRENGTH_FADING = 0.2     # 衰退
STRENGTH_FORGOTTEN = 0.05 # 遗忘

# 默认稳定性系数（轮次）
DEFAULT_STABILITY = 10.0


def calculate_strength(
    current_turn: int,
    memory_turn: int,
    access_count: int = 1,
    stability: float = DEFAULT_STABILITY,
) -> float:
    """
    基于对话轮次计算记忆强度

    参数:
        current_turn: 当前对话轮次
        memory_turn: 记忆创建/访问时的轮次
        access_count: 访问次数
        stability: 稳定性系数（轮次），越大遗忘越慢

    返回:
        记忆强度（0.0 ~ 1.0）
    """
    # 距离（轮次）
    turns_ago = current_turn - memory_turn

    if turns_ago < 0:
        turns_ago = 0

    # 稳定性随访问次数增加（对数增长）
    adjusted_stability = stability * (1 + math.log1p(access_count))

    # 遗忘公式：strength = e^(-turns_ago / adjusted_stability)
    strength = math.exp(-turns_ago / adjusted_stability)

    return max(0.0, min(1.0, strength))


def get_strength_level(strength: float) -> str:
    """获取强度等级名称"""
    if strength >= STRENGTH_ALIVE:
        return "alive"
    elif strength >= STRENGTH_NORMAL:
        return "normal"
    elif strength >= STRENGTH_FADING:
        return "fading"
    else:
        return "forgotten"


def build_memory_metadata(
    memory_type: str = "knowledge",
    current_turn: int = 0,
    stability: float = DEFAULT_STABILITY,
) -> dict[str, Any]:
    """
    构建记忆元数据

    参数:
        memory_type: 记忆类型（conversation/preference/knowledge）
        current_turn: 创建时的轮次
        stability: 稳定性系数

    返回:
        元数据字典
    """
    return {
        "created_at_turn": current_turn,
        "last_accessed_turn": current_turn,
        "access_count": 1,
        "stability": stability,
        "strength": 1.0,
        "strength_level": "alive",
        "memory_type": memory_type,
    }


def update_on_access(metadata: dict[str, Any], current_turn: int) -> dict[str, Any]:
    """
    访问时更新记忆元数据

    参数:
        metadata: 现有元数据
        current_turn: 当前轮次

    返回:
        更新后的元数据
    """
    metadata["last_accessed_turn"] = current_turn
    metadata["access_count"] = metadata.get("access_count", 0) + 1

    # 重新计算强度（访问后强度最高）
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=current_turn,
        access_count=metadata["access_count"],
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )

    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)

    return metadata


def decay_memory(metadata: dict[str, Any], current_turn: int) -> dict[str, Any]:
    """
    计算记忆当前衰减后的强度（不更新访问时间）

    参数:
        metadata: 现有元数据
        current_turn: 当前轮次

    返回:
        更新强度后的元数据
    """
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )

    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)

    return metadata


def should_cleanup(
    metadata: dict[str, Any],
    current_turn: int,
    min_strength: float = STRENGTH_FORGOTTEN,
) -> bool:
    """判断记忆是否应该被清理"""
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    return strength < min_strength
```

def add_memory(
    self,
    content: str,
    memory_type: str = "knowledge",
    metadata: dict | None = None,
    current_turn: int = 0,
    stability: float = DEFAULT_STABILITY,
) -> str:
    """
    添加记忆

    参数:
        content: 记忆内容
        memory_type: 记忆类型（conversation/preference/knowledge）
        metadata: 额外元数据
        current_turn: 当前对话轮次
        stability: 稳定性系数（轮次），越大遗忘越慢

    返回:
        记忆 ID
    """
    if metadata is None:
        metadata = {}

    decay_meta = build_memory_metadata(memory_type, current_turn, stability)
    metadata.update(decay_meta)

    memory_id = str(uuid.uuid4())
    self.collection.add(
        ids=[memory_id],
        documents=[content],
        metadatas=[metadata],
    )

    logger.info(f"添加记忆: {memory_id[:8]}... (turn={current_turn}, stability={stability})")
    return memory_id

### 2.3 修改 search 方法

找到 `search` 方法，替换为：

```python
def search(
    self,
    query: str,
    n_results: int = 5,
    memory_type: str | None = None,
    current_turn: int = 0,
    min_strength: float = 0.0,
) -> list[dict]:
    """
    搜索记忆（按语义相似度 + 记忆强度加权排序）

    参数:
        query: 搜索查询
        n_results: 返回结果数量
        memory_type: 过滤记忆类型
        current_turn: 当前对话轮次
        min_strength: 最低强度阈值

    返回:
        记忆列表，按综合分数排序
    """
    where = {}
    if memory_type:
        where["memory_type"] = memory_type

    results = self.collection.query(
        query_texts=[query],
        n_results=n_results * 3,
        where=where if where else None,
    )

    if not results["ids"] or not results["ids"][0]:
        return []

    memories = []
    for i in range(len(results["ids"][0])):
        metadata = results["metadatas"][0][i] if results["metadatas"] else {}

        # 衰减计算
        if "last_accessed_turn" in metadata:
            metadata = decay_memory(metadata, current_turn)
            strength = metadata.get("strength", 1.0)
        else:
            strength = 1.0

        # 过滤低强度记忆
        if strength < min_strength:
            continue

        # 综合分数 = 语义相似度 * (0.3 + 0.7 * 强度)
        distance = results["distances"][0][i] if results["distances"] else 1.0
        similarity = max(0, 1 - distance)
        combined_score = similarity * (0.3 + 0.7 * strength)

        memories.append({
            "id": results["ids"][0][i],
            "content": results["documents"][0][i],
            "metadata": metadata,
            "similarity": round(similarity, 3),
            "strength": round(strength, 3),
            "strength_level": get_strength_level(strength),
            "combined_score": round(combined_score, 3),
        })

    memories.sort(key=lambda x: x["combined_score"], reverse=True)

    return memories[:n_results]
```

### 2.4 添加 access_memory 方法

在 `search` 方法后添加：

```python
def access_memory(self, memory_id: str, current_turn: int = 0) -> dict | None:
    """
    访问记忆（加强记忆强度）

    参数:
        memory_id: 记忆 ID
        current_turn: 当前对话轮次

    返回:
        更新后的记忆，或 None
    """
    results = self.collection.get(ids=[memory_id])

    if not results["ids"]:
        return None

    metadata = results["metadatas"][0] if results["metadatas"] else {}

    if "last_accessed_turn" in metadata:
        metadata = update_on_access(metadata, current_turn)
        self.collection.update(ids=[memory_id], metadatas=[metadata])

    return {
        "id": memory_id,
        "content": results["documents"][0],
        "metadata": metadata,
    }
```

def cleanup_forgotten(
    self,
    current_turn: int,
    min_strength: float = 0.05,
) -> dict[str, Any]:
    """
    清理遗忘的记忆

    参数:
        current_turn: 当前对话轮次
        min_strength: 最低保留强度

    返回:
        {"deleted": int, "total": int}
    """
    all_items = self.collection.get()

    if not all_items["ids"]:
        return {"deleted": 0, "total": 0}

    to_delete = []
    for i, memory_id in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}

        if "last_accessed_turn" not in metadata:
            continue

        strength = calculate_strength(
            current_turn=current_turn,
            memory_turn=metadata["last_accessed_turn"],
            access_count=metadata.get("access_count", 1),
            stability=metadata.get("stability", DEFAULT_STABILITY),
        )

        if strength < min_strength:
            to_delete.append(memory_id)

    if to_delete:
        self.collection.delete(ids=to_delete)
        logger.info(f"清理遗忘记忆: {len(to_delete)} 条")

    return {"deleted": len(to_delete), "total": len(all_items["ids"])}
```

def get_memory_stats(self, current_turn: int = 0) -> dict[str, Any]:
    """获取记忆统计（含强度分布）"""
    all_items = self.collection return {"total": 0, "by_type": {}, "by_strength": {}}

    stats = {
        "total": len(all_items["ids"]),
        "by_type": {},
        "by_strength": {"alive": 0, "normal": 0, "fading": 0, "forgotten": 0},
    }

    for i in range(len(all_items["ids"])):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        mem_type = metadata.get("memory_type", "unknown")
        stats["by_type"][mem_type] = stats["by_type"].get(mem_type, 0) + 1

        if "last_accessed_turn" in metadata:
            strength = calculate_strength(
                current_turn=current_turn,
                memory_turn=metadata["last_accessed_turn"],
                access_count=metadata.get("access_count", 1),
                stability=metadata.get("stability", DEFAULT_STABILITY),
            )
            level = get_strength_level(strength)
            stats["by_strength"][level] = stats["by_strength"].get(level, 0) + 1

    return stats
```
