```python
#!/usr/bin/env python3
"""
记忆衰减模块 - 基于对话轮次的艾宾浩斯遗忘曲线

核心逻辑：
    对 AI 来说"时间"= 对话轮次，不是日历时间
    记忆强度随距离当前轮次衰减
    每次访问（复习）会加强记忆

Author: Plector
Version: 1.0.0
"""

import math
import logging
from typing import Any

logger = logging.getLogger(__name__)

# 强度等级阈值
STRENGTH_ALIVE = 0.8      # 鲜活
STRENGTH_NORMAL = 0.5     # 正常
STRENGTH_FADING = 0.2     # 衰退
STRENGTH_FORGOTTEN = 0.05 # 遗忘

# 默认稳定性系数（轮次）
DEFAULT_STABILITY = 10.0


def calculate_strength(
    current_turn: int,
    memory_turn: int,
    access_count: int = 1,
    stability: float = DEFAULT_STABILITY,
) -> float:
    """
    基于对话轮次计算记忆强度

    参数:
        current_turn: 当前对话轮次
        memory_turn: 记忆创建/访问时的轮次
        access_count: 访问次数
        stability: 稳定性系数（轮次），越大遗忘越慢

    返回:
        记忆强度（0.0 ~ 1.0）
    """
    turns_ago = max(0, current_turn - memory_turn)
    adjusted_stability = stability * (1 + math.log1p(access_count))
    strength = math.exp(-turns_ago / adjusted_stability)
    return max(0.0, min(1.0, strength))


def get_strength_level(strength: float) -> str:
    """获取强度等级名称"""
    if strength >= STRENGTH_ALIVE:
        return "alive"
    elif strength >= STRENGTH_NORMAL:
        return "normal"
    elif strength >= STRENGTH_FADING:
        return "fading"
    else:
        return "forgotten"


def build_memory_metadata(
    memory_type: str = "knowledge",
    current_turn: int = 0,
    stability: float = DEFAULT_STABILITY,
) -> dict[str, Any]:
    """构建记忆元数据"""
    return {
        "created_at_turn": current_turn,
        "last_accessed_turn": current_turn,
        "access_count": 1,
        "stability": stability,
        "strength": 1.0,
        "strength_level": "alive",
        "memory_type": memory_type,
        "associations": "",
        "scenarios": "",
    }


def update_on_access(metadata: dict[str, Any], current_turn: int) -> dict[str, Any]:
    """访问时更新记忆元数据"""
    metadata["last_accessed_turn"] = current_turn
    metadata["access_count"] = metadata.get("access_count", 0) + 1
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=current_turn,
        access_count=metadata["access_count"],
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)
    return metadata


def decay_memory(metadata: dict[str, Any], current_turn: int) -> dict[str, Any]:
    """计算记忆当前衰减后的强度"""
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    metadata["strength"] = strength
    metadata["strength_level"] = get_strength_level(strength)
    return metadata


def should_cleanup(
    metadata: dict[str, Any],
    current_turn: int,
    min_strength: float = STRENGTH_FORGOTTEN,
) -> bool:
    """判断记忆是否应该被清理"""
    strength = calculate_strength(
        current_turn=current_turn,
        memory_turn=metadata["last_accessed_turn"],
        access_count=metadata.get("access_count", 1),
        stability=metadata.get("stability", DEFAULT_STABILITY),
    )
    return strength < min_strength
```

```python
#!/usr/bin/env python3
"""
记忆联想模块 - 6 种联想模式

联想模式：
    1. 语义相似 - ChromaDB 向量检索（已有）
    2. 触景生情 - 用当前上下文关键词触发
    3. 触类旁通 - 用关联概念扩展检索
    4. 时序联想 - 同一轮次附近的记忆
    5. 温故知新 - 定期重访旧记忆
    6. 举一反三 - 存储时生成适用场景

Author: Plector
Version: 1.0.0
"""

import logging
from typing import Any

logger = logging.getLogger(__name__)


def search_by_triggers(
    collection,
    query_keywords: list[str],
    current_turn: int,
    n_results: int = 5,
) -> list[dict]:
    """
    触景生情：用关键词触发相关记忆

    参数:
        collection: ChromaDB collection
        query_keywords: 从当前上下文提取的关键词列表
        current_turn: 当前轮次
        n_results: 返回数量
    """
    if not query_keywords:
        return []

    results = []
    seen_ids = set()

    for keyword in query_keywords[:5]:
        try:
            hits = collection.query(
                query_texts=[keyword],
                n_results=3,
            )
            if hits["ids"] and hits["ids"][0]:
                for i, mid in enumerate(hits["ids"][0]):
                    if mid not in seen_ids:
                        seen_ids.add(mid)
                        results.append({
                            "id": mid,
                            "content": hits["documents"][0][i],
                            "metadata": hits["metadatas"][0][i] if hits["metadatas"] else {},
                            "source": "trigger",
                            "trigger_keyword": keyword,
                        })
        except Exception:
            continue

    return results[:n_results]


def search_by_associations(
    collection,
    query: str,
    current_turn: int,
    n_results: int = 5,
) -> list[dict]:
    """
    触类旁通：用关联概念扩展检索

    先找到语义相关的记忆，再用其 associations 字段检索更多相关记忆
    """
    # 第一轮：语义检索
    initial = collection.query(query_texts=[query], n_results=3)
    if not initial["ids"] or not initial["ids"][0]:
        return []

    # 提取关联概念
    association_keywords = []
    for metadata in (initial["metadatas"][0] if initial["metadatas"] else []):
        assoc = metadata.get("associations", "")
        if assoc:
            association_keywords.extend([a.strip() for a in assoc.split(",") if a.strip()])

    if not association_keywords:
        return []

    # 第二轮：用关联概念检索
    results = []
    seen_ids = set(initial["ids"][0])

    for keyword in association_keywords[:5]:
        try:
            hits = collection.query(query_texts=[keyword], n_results=2)
            if hits["ids"] and hits["ids"][0]:
                for i, mid in enumerate(hits["ids"][0]):
                    if mid not in seen_ids:
                        seen_ids.add(mid)
                        results.append({
                            "id": mid,
                            "content": hits["documents"][0][i],
                            "metadata": hits["metadatas"][0][i] if hits["metadatas"] else {},
                            "source": "association",
                            "via_keyword": keyword,
                        })
        except Exception:
            continue

    return results[:n_results]


def search_by_temporal(
    collection,
    current_turn: int,
    window: int = 5,
    n_results: int = 5,
) -> list[dict]:
    """
    时序联想：检索当前轮次附近的记忆

    参数:
        collection: ChromaDB collection
        current_turn: 当前轮次
        window: 前后窗口大小（轮次）
        n_results: 返回数量
    """
    try:
        all_items = collection.get()
    except Exception:
        return []

    if not all_items["ids"]:
        return []

    results = []
    for i, mid in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        created_turn = metadata.get("created_at_turn", 0)

        if abs(current_turn - created_turn) <= window:
            results.append({
                "id": mid,
                "content": all_items["documents"][i],
                "metadata": metadata,
                "source": "temporal",
                "turns_ago": current_turn - created_turn,
            })

    results.sort(key=lambda x: abs(x["turns_ago"]))
    return results[:n_results]


def search_by_review(
    collection,
    current_turn: int,
    review_interval: int = 50,
    n_results: int = 5,
) -> list[dict]:
    """
    温故知新：定期重访旧记忆

    每隔 review_interval 轮，检索最久未访问的记忆，触发重新理解

    参数:
        collection: ChromaDB collection
        current_turn: 当前轮次
        review_interval: 审查间隔（轮次）
        n_results: 返回数量
    """
    # 只在特定轮次触发
    if current_turn % review_interval != 0:
        return []

    try:
        all_items = collection.get()
    except Exception:
        return []

    if not all_items["ids"]:
        return []

    results = []
    for i, mid in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        last_accessed = metadata.get("last_accessed_turn", 0)

        results.append({
            "id": mid,
            "content": all_items["documents"][i],
            "metadata": metadata,
            "source": "review",
            "turns_since_access": current_turn - last_accessed,
        })

    results.sort(key=lambda x: x["turns_since_access"], reverse=True)
    return results[:n_results]


def merge_and_rank(
    *result_lists: list[dict],
    current_turn: int = 0,
    n_results: int = 10,
) -> list[dict]:
    """
    合并多路召回结果，去重并按综合分数排序

    权重：
        语义相似: 0.4
        触景生情: 0.3
        触类旁通: 0.2
        时序联想: 0.15
        温故知新: 0.1
    """
    source_weights = {
        "semantic": 0.4,
        "trigger": 0.3,
        "association": 0.2,
        "temporal": 0.15,
        "review": 0.1,
    }

    merged = {}
    for results in result_lists:
        for item in results:
            mid = item["id"]
            if mid not in merged:
                merged[mid] = item
                merged[mid]["recall_sources"] = [item.get("source", "semantic")]
            else:
                if item.get("source") not in merged[mid]["recall_sources"]:
                    merged[mid]["recall_sources"].append(item.get("source", "semantic"))

    # 计算综合分数
    for mid, item in merged.items():
        sources = item["recall_sources"]
        source_score = sum(source_weights.get(s, 0.1) for s in sources)
        strength = item.get("metadata", {}).get("strength", 1.0)
        item["combined_score"] = round(source_score * (0.3 + 0.7 * strength), 3)

    ranked = sorted(merged.values(), key=lambda x: x["combined_score"], reverse=True)
    return ranked[:n_results]
```

### 3.3 修改 search 方法

```python
def search(
    self,
    query: str,
    n_results: int = 5,
    memory_type: str | None = None,
    current_turn: int = 0,
    min_strength: float = 0.0,
    use_association: bool = True,
) -> list[dict]:
    """
    多路召回搜索记忆

    联想模式：
        1. 语义相似 - 向量检索
        2. 触景生情 - 关键词触发
        3. 触类旁通 - 关联概念扩展
        4. 时序联想 - 附近轮次记忆
        5. 温故知新 - 定期重访旧记忆

    参数:
        query: 搜索查询
        n_results: 返回结果数量
        memory_type: 过滤记忆类型
        current_turn: 当前对话轮次
        min_strength: 最低强度阈值
        use_association: 是否启用联想模式
    """
    where = {}
    if memory_type:
        where["memory_type"] = memory_type

    # 1. 语义相似检索
    semantic_results = self._semantic_search(query, where, current_turn, n_results * 2)

    if not use_association:
        return self._filter_by_strength(semantic_results, min_strength)[:n_results]

    # 提取关键词用于触景生情
    keywords = self._extract_keywords(query)

    # 2. 触景生情
    trigger_results = search_by_triggers(
        self.collection, keywords, current_turn, n_results
    )

    # 3. 触类旁通
    association_results = search_by_associations(
        self.collection, query, current_turn, n_results
    )

    # 4. 时序联想
    temporal_results = search_by_temporal(
        self.collection, current_turn, window=5, n_results=n_results
    )

    # 5. 温故知新
    review_results = search_by_review(
        self.collection, current_turn, review_interval=50, n_results=n_results
    )

    # 合并排序
    all_results = merge_and_rank(
        semantic_results,
        trigger_results,
        association_results,
        temporal_results,
        review_results,
        current_turn=current_turn,
        n_results=n_results * 2,
    )

    # 过滤低强度
    filtered = self._filter_by_strength(all_results, min_strength)

    return filtered[:n_results]

    def _semantic_search(self, query, where, current_turn, n_results):
        """语义相似检索"""
        results = self.collection.query(
            query_texts=[query],
            n_results=n_results,
            where=where if where else None,
        )

        if not results["ids"] or not results["ids"][0]:
            return []

        memories = []
        for i in range(len(results["ids"][0])):
            metadata = results["metadatas"][0][i] if results["metadatas"] else {}

            if "last_accessed_turn" in metadata:
                metadata = decay_memory(metadata, current_turn)
                strength = metadata.get("strength", 1.0)
            else:
                strength = 1.0

            distance = results["distances"][0][i] if results["distances"] else 1.0
            similarity = max(0, 1 - distance)

            memories.append({
                "id": results["ids"][0][i],
                "content": results["documents"][0][i],
                "metadata": metadata,
                "similarity": round(similarity, 3),
                "strength": round(strength, 3),
                "strength_level": get_strength_level(strength),
                "source": "semantic",
                "combined_score": round(similarity * (0.3 + 0.7 * strength), 3),
            })

        return memories

    def _filter_by_strength(self, memories, min_strength):
        """按强度过滤"""
        filtered = [m for m in memories if m.get("strength", 1.0) >= min_strength]
        filtered.sort(key=lambda x: x.get("combined_score", 0), reverse=True)
        return filtered

    @staticmethod
    def _extract_keywords(query: str) -> list[str]:
        """从查询中提取关键词"""
        import re
        # 简单分词：按空格和标点分割，过滤短词
        words = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+', query)
        return [w for w in words if len(w) >= 2][:5]
```

```python
def cleanup_forgotten(
    self,
    current_turn: int,
    min_strength: float = 0.05,
) -> dict[str, Any]:
    """清理遗忘的记忆"""
    all_items = self.collection.get()

    if not all_items["ids"]:
        return {"deleted": 0, "total": 0}

    to_delete = []
    for i, mid in enumerate(all_items["ids"]):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        if "last_accessed_turn" not in metadata:
            continue

        strength = calculate_strength(
            current_turn=current_turn,
            memory_turn=metadata["last_accessed_turn"],
            access_count=metadata.get("access_count", 1),
            stability=metadata.get("stability", DEFAULT_STABILITY),
        )

        if strength < min_strength:
            to_delete.append(mid)

    if to_delete:
        self.collection.delete(ids=to_delete)
        logger.info(f"清理    self.collection.update(ids=[memory_id], metadatas遗忘记忆: {len(to_delete)} 条")

    return {"deleted": len(to_delete), "total": len(all_items["ids"])}
```

```python
def get_memory_stats(self, current_turn: int = 0) -> dict[str, Any]:
    """获取记忆统计（含强度分布）"""
    all_items = self.collection.get()

    if not all_items["ids"]:
        return {"total": 0, "by_type": {}, "by_strength": {}}

    stats = {
        "total": len(all_items["ids"]),
        "by_type": {},
        "by_strength": {"alive": 0, "normal": 0, "fading": 0, "forgotten": 0},
    }

    for i in range(len(all_items["ids"])):
        metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
        mem_type = metadata.get("memory_type", "unknown")
        stats["by_type"][mem_type] = stats["by_type"].get(mem_type, 0) + 1

        if "last_accessed_turn" in metadata:
            strength = calculate_strength(
                current_turn=current_turn,
                memory_turn=metadata["last_accessed_turn"],
                access_count=metadata.get("access_count", 1),
                stability=metadata.get("stability", DEFAULT_STABILITY),
            )
            level = get_strength_level(strength)
            stats["by_strength"][level] = stats["by_strength"].get(level, 0) + 1

    return stats
```
