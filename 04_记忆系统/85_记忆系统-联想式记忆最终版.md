    def _calculate_strength(self, current_turn: int, memory_turn: int,
                            access_count: int = 1, stability: float = None) -> float:
        """基于对话轮次计算记忆强度"""
        if stability is None:
            stability = self.DEFAULT_STABILITY
        turns_ago = max(0, current_turn - memory_turn)
        adjusted = stability * (1 + math.log1p(access_count))
        return max(0.0, min(1.0, math.exp(-turns_ago / adjusted)))

    def _get_strength_level(self, strength: float) -> str:
        """获取强度等级"""
        if strength >= self.STRENGTH_ALIVE:
            return "alive"
        elif strength >= self.STRENGTH_NORMAL:
            return "normal"
        elif strength >= self.STRENGTH_FADING:
            return "fading"
        return "forgotten"

    def _build_decay_metadata(self, memory_type: str, current_turn: int,
                               stability: float = None) -> dict:
        """构建含衰减信息的元数据"""
        if stability is None:
            stability = self.DEFAULT_STABILITY
        return {
            "memory_type": memory_type,
            "created_at_turn": current_turn,
            "last_accessed_turn": current_turn,
            "access_count": 1,
            "stability": stability,
            "strength": 1.0,
            "strength_level": "alive",
            "associations": "",
            "scenarios": "",
            "opposites": "",
            "reinterpreted": False,
        }

    def _update_on_access(self, metadata: dict, current_turn: int) -> dict:
        """访问时更新衰减信息"""
        metadata["last_accessed_turn"] = current_turn
        metadata["access_count"] = metadata.get("access_count", 0) + 1
        strength = self._calculate_strength(
            current_turn, current_turn,
            metadata["access_count"],
            metadata.get("stability", self.DEFAULT_STABILITY),
        )
        metadata["strength"] = strength
        metadata["strength_level"] = self._get_strength_level(strength)
        return metadata

    def _decay_memory(self, metadata: dict, current_turn: int) -> dict:
        """计算当前衰减强度（不更新访问时间）"""
        strength = self._calculate_strength(
            current_turn,
            metadata.get("last_accessed_turn", 0),
            metadata.get("access_count", 1),
            metadata.get("stability", self.DEFAULT_STABILITY),
        )
        metadata["strength"] = strength
        metadata["strength_level"] = self._get_strength_level(strength)
        return metadata
```

    def _extract_keywords(self, query: str) -> list[str]:
        """从查询中提取关键词"""
        words = re.findall(r'[\u4e00-\u9fff]+|[a-zA-Z]+', query)
        return [w for w in words if len(w) >= 2][:5]

    def _semantic_search(self, query: str, where: dict,
                          current_turn: int, n_results: int) -> list[dict]:
        """语义相似检索"""
        results = self.collection.query(
            query_texts=[query],
            n_results=n_results,
            where=where if where else None,
            include=["documents", "metadatas", "distances"],
        )
        if not results["ids"] or not results["ids"][0]:
            return []

        memories = []
        for i in range(len(results["ids"][0])):
            metadata = results["metadatas"][0][i] if results["metadatas"] else {}
            if "last_accessed_turn" in metadata:
                metadata = self._decay_memory(metadata, current_turn)
                strength = metadata.get("strength", 1.0)
            else:
                strength = 1.0

            distance = results["distances"][0][i] if results["distances"] else 1.0
            similarity = max(0, 1 - distance)

            memories.append({
                "id": results["ids"][0][i],
                "content": results["documents"][0][i],
                "metadata": metadata,
                "similarity": round(similarity, 3),
                "strength": round(strength, 3),
                "strength_level": self._get_strength_level(strength),
                "source": "semantic",
                "combined_score": round(similarity * (0.3 + 0.7 * strength), 3),
            })
        return memories

    def _search_by_triggers(self, keywords: list[str],
                             current_turn: int, n_results: int) -> list[dict]:
        """触景生情：关键词触发"""
        if not keywords:
            return []
        results, seen = [], set()
        for kw in keywords[:5]:
            try:
                hits = self.collection.query(
                    query_texts=[kw],
                    n_results=3,
                    include=["documents", "metadatas", "distances"],
                )
                if hits["ids"] and hits["ids"][0] and hits["documents"]:
                    for i, mid in enumerate(hits["ids"][0]):
                        if mid not in seen:
                            seen.add(mid)
                            metadata = hits["metadatas"][0][i] if hits["metadatas"] else {}
                            if "last_accessed_turn" in metadata:
                                metadata = self._decay_memory(metadata, current_turn)
                            results.append({
                                "id": mid,
                                "content": hits["documents"][0][i],
                                "metadata": metadata,
                                "source": "trigger",
                                "trigger_keyword": kw,
                            })
            except Exception as e:
                logger.warning(f"触景检索异常: {e}")
                continue
        return results[:n_results]

    def _search_by_associations(self, query: str,
                                 current_turn: int, n_results: int) -> list[dict]:
        """触类旁通：关联概念扩展"""
        initial = self.collection.query(
            query_texts=[query],
            n_results=3,
            include=["documents", "metadatas", "distances"],
        )
        if not initial["ids"] or not initial["ids"][0]:
            return []

        assoc_keywords = []
        for m in (initial["metadatas"][0] if initial["metadatas"] else []):
            a = m.get("associations", "")
            if a:
                assoc_keywords.extend([x.strip() for x in a.split(",") if x.strip()])

        if not assoc_keywords:
            return []

        results, seen = [], set(initial["ids"][0])
        for kw in assoc_keywords[:5]:
            try:
                hits = self.collection.query(
                    query_texts=[kw],
                    n_results=2,
                    include=["documents", "metadatas", "distances"],
                )
                if hits["ids"] and hits["ids"][0] and hits["documents"]:
                    for i, mid in enumerate(hits["ids"][0]):
                        if mid not in seen:
                            seen.add(mid)
                            metadata = hits["metadatas"][0][i] if hits["metadatas"] else {}
                            if "last_accessed_turn" in metadata:
                                metadata = self._decay_memory(metadata, current_turn)
                            results.append({
                                "id": mid,
                                "content": hits["documents"][0][i],
                                "metadata": metadata,
                                "source": "association",
                                "via_keyword": kw,
                            })
            except Exception as e:
                logger.warning(f"关联检索异常: {e}")
                continue
        return results[:n_results]

    def _search_by_temporal(self, current_turn: int,
                             window: int = 5, n_results: int = 5) -> list[dict]:
        """时序联想：附近轮次的记忆"""
        try:
            all_items = self.collection.get()
        except Exception:
            return []

        if not all_items["ids"]:
            return []

        results = []
        for i, mid in enumerate(all_items["ids"]):
            metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
            created = metadata.get("created_at_turn")
            if created is not None and abs(created - current_turn) <= window:
                if "last_accessed_turn" in metadata:
                    metadata = self._decay_memory(metadata, current_turn)
                results.append({
                    "id": mid,
                    "content": all_items["documents"][i],
                    "metadata": metadata,
                    "source": "temporal",
                    "turns_ago": current_turn - created,
                })

        results.sort(key=lambda x: abs(x["turns_ago"]))
        return results[:n_results]

    def _search_by_review(self, current_turn: int,
                           review_interval: int = 50, n_results: int = 5) -> list[dict]:
        """温故知新：定期重访旧记忆"""
        if current_turn % review_interval != 0:
            return []

        try:
            all_items = self.collection.get()
        except Exception:
            return []

        if not all_items["ids"]:
            return []

        results = []
        for i, mid in enumerate(all_items["ids"]):
            metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
            last_accessed = metadata.get("last_accessed_turn", 0)
            if "last_accessed_turn" in metadata:
                metadata = self._decay_memory(metadata, current_turn)
            results.append({
                "id": mid,
                "content": all_items["documents"][i],
                "metadata": metadata,
                "source": "review",
                "turns_since_access": current_turn - last_accessed,
            })

        results.sort(key=lambda x: x["turns_since_access"], reverse=True)
        return results[:n_results]

    def _merge_and_rank(self, *result_lists: list[dict],
                         current_turn: int = 0, n_results: int = 10) -> list[dict]:
        """合并多路召回结果，去重并按综合分数排序"""
        merged = {}
        for results in result_lists:
            for item in results:
                mid = item["id"]
                src = item.get("source", "semantic")
                if mid not in merged:
                    merged[mid] = item.copy()
                    merged[mid]["recall_sources"] = [src]
                else:
                    if src not in merged[mid]["recall_sources"]:
                        merged[mid]["recall_sources"].append(src)

        for mid, item in merged.items():
            src_score = min(1.0, sum(
                self.SOURCE_WEIGHTS.get(s, 0.1) for s in item["recall_sources"]
            ))
            strength = item.get("metadata", {}).get("strength", 1.0)
            item["combined_score"] = round(src_score * (0.3 + 0.7 * strength), 3)

        ranked = sorted(merged.values(), key=lambda x: x["combined_score"], reverse=True)
        return ranked[:n_results]
```

    async def _call_llm_json(self, prompt: str, llm_client) -> dict:
        """调用 LLM 生成 JSON"""
        try:
            result = await llm_client.chat(
                messages=[{"role": "user", "content": prompt}],
                max_tokens=500,
            )
            text = result.get("content", "") if isinstance(result, dict) else str(result)
            text = text.strip()
            if text.startswith("```"):
                text = text.split("\n", 1)[1].rsplit("```", 1)[0]
            return json.loads(text)
        except Exception as e:
            logger.error(f"LLM 调用失败: {e}", exc_info=True)
            return {}

    async def _generate_scenarios(self, content: str, llm_client) -> dict:
        """举一反三：LLM 生成适用场景和关联"""
        prompt = f"""分析以下内容，生成适用场景和关联概念。

内容：
{content}

请以 JSON 格式返回：
{{
  "associations": "关联概念，逗号分隔，3-8个",
  "scenarios": "适用场景，逗号分隔，2-5个",
  "opposites": "相反或对比概念，逗号分隔，1-3个",
  "summary": "一句话总结"
}}

只返回 JSON，不要其他内容。如果内容太少无法分析，返回空字符串。"""
        return await self._call_llm_json(prompt, llm_client)

    async def _reinterpret_memory(self, memory: dict,
                                   current_context: str, llm_client) -> dict:
        """温故知新：LLM 重新理解旧记忆"""
        prompt = f"""分析以下旧记忆和当前上下文，发现新的关联。

旧记忆内容：
{memory.get("content", "")}

当前上下文：
{current_context}

请以 JSON 格式返回：
{{
  "new_associations": "新发现的关联概念，逗号分隔",
  "new_scenarios": "新发现的适用场景，逗号分隔",
  "insight": "新的理解或洞察，一句话",
  "has_changed": true/false
}}

只返回 JSON，不要其他内容。"""
        result = await self._call_llm_json(prompt, llm_client)

        if result and result.get("has_changed"):
            metadata = memory.get("metadata", {})
            for field, key in [("associations", "new_associations"),
                                ("scenarios", "new_scenarios")]:
                new_val = result.get(key, "")
                if new_val:
                    existing = metadata.get(field, "")
                    merged = list(set(
                        [x.strip() for x in (existing + "," + new_val).split(",") if x.strip()]
                    ))
                    metadata[field] = ",".join(merged)
            metadata["reinterpreted"] = True
            memory["metadata"] = metadata

        return memory

    async def _synthesize_new(self, memories: list[dict],
                               task: str, llm_client) -> dict:
        """推陈出新：融合多条记忆生成新方案"""
        texts = [f"[记忆{i+1}] {m.get('content', '')}" for i, m in enumerate(memories)]
        prompt = f"""基于以下记忆和任务，融合生成一个新方案。

任务：{task}

相关记忆：
{chr(10).join(texts)}

请以 JSON 格式返回：
{{
  "new_solution": "融合后的新方案",
  "based_on": "基于哪些记忆",
  "key_insights": "关键洞察，逗号分隔"
}}

只返回 JSON，不要其他内容。"""
        return await self._call_llm_json(prompt, llm_client)

    async def _detect_conflicts(self, new_content: str,
                                 existing: list[dict], llm_client) -> list[dict]:
        """恍然大悟：检测新旧记忆矛盾"""
        if not existing:
            return []
        texts = [f"[记忆{i+1}] {m.get('content', '')}" for i, m in enumerate(existing)]
        prompt = f"""检测以下新内容是否与已有记忆矛盾。

新内容：
{new_content}

已有记忆：
{chr(10).join(texts)}

请以 JSON 格式返回：
{{
  "has_conflicts": true/false,
  "conflicts": [
    {{
      "memory_index": 1,
      "conflict_description": "矛盾描述",
      "resolution": "建议如何解决"
    }}
  ]
}}

只返回 JSON。没有矛盾则返回 {{"has_conflicts": false, "conflicts": []}}。"""
        result = await self._call_llm_json(prompt, llm_client)

        if result and result.get("has_conflicts"):
            conflicts = []
            for c in result.get("conflicts", []):
                idx = c.get("memory_index", 1) - 1
                if 0 <= idx < len(existing):
                    conflicts.append({
                        "memory_id": existing[idx].get("id"),
                        "memory_content": existing[idx].get("content", "")[:100],
                        "description": c.get("conflict_description", ""),
                        "resolution": c.get("resolution", ""),
                    })
            return conflicts
        return []
```

    def search(self, query: str, n_results: int = 5, memory_type: str = None,
               current_turn: int = 0, min_strength: float = 0.0,
               use_association: bool = True) -> list[dict]:
        """
        多路召回搜索记忆

        联想模式：
            1. 语义相似 - 向量检索
            2. 触景生情 - 关键词触发
            3. 触类旁通 - 关联概念扩展
            4. 时序联想 - 附近轮次记忆
            5. 温故知新 - 定期重访旧记忆
        """
        where = {}
        if memory_type:
            where["memory_type"] = memory_type

        semantic = self._semantic_search(query, where, current_turn, n_results * 2)

        if not use_association:
            filtered = [m for m in semantic if m.get("strength", 1.0) >= min_strength]
            return filtered[:n_results]

        keywords = self._extract_keywords(query)
        trigger = self._search_by_triggers(keywords, current_turn, n_results)
        association = self._search_by_associations(query, current_turn, n_results)
        temporal = self._search_by_temporal(current_turn, window=5, n_results=n_results)
        review = self._search_by_review(current_turn, review_interval=50, n_results=n_results)

        all_results = self._merge_and_rank(
            semantic, trigger, association, temporal, review,
            current_turn=current_turn, n_results=n_results * 2,
        )

        filtered = [m for m in all_results if m.get("strength", 1.0) >= min_strength]
        filtered.sort(key=lambda x: x.get("combined_score", 0), reverse=True)
        return filtered[:n_results]
```

    async def search_with_llm(self, query: str, task: str = "",
                               n_results: int = 5, current_turn: int = 0,
                               min_strength: float = 0.1, llm_client=None) -> dict:

            "reinterpreted": reinterpreted,

        }

    def cleanup_forgotten(self, current_turn: int,
                           min_strength: float = 0.05) -> dict:
        """清理遗忘的记忆"""
        all_items = self.collection.get()
        if not all_items["ids"]:
            return {"deleted": 0, "total": 0}
        to_delete = []
        for i, mid in enumerate(all_items["ids"]):
            metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
            if "last_accessed_turn" not in metadata:
                continue
            strength = self._calculate_strength(
                current_turn, metadata["last_accessed_turn"],
                metadata.get("access_count", 1),
                metadata.get("stability", self.DEFAULT_STABILITY),
            )
            if strength < min_strength:
                to_delete.append(mid)
        if to_delete:
            self.collection.delete(ids=to_delete)
            logger.info(f"清理遗忘记忆: {len(to_delete)} 条")
        return {"deleted": len(to_delete), "total": len(all_items["ids"])}
```

        """获取记忆统计"""
        all_items = self.collection.get()
        if not all_items["ids"]:
            return {"total": 0, "by_type": {}, "by_strength": {}}
        stats = {
            "total": len(all_items["ids"]),
            "by_type": {},
            "by_strength": {"alive": 0, "normal": 0, "fading": 0, "forgotten": 0},
        }
        for i in range(len(all_items["ids"])):
            metadata = all_items["metadatas"][i] if all_items["metadatas"] else {}
            mt = metadata.get("memory_type", "unknown")
            stats["by_type"][mt] = stats["by_type"].get(mt, 0) + 1
            if "last_accessed_turn" in metadata:
                strength = self._calculate_strength(
                    current_turn, metadata["last_accessed_turn"],
                    metadata.get("access_count", 1),
                    metadata.get("stability", self.DEFAULT_STABILITY),
                )
                level = self._get_strength_level(strength)
                stats["by_strength"][level] = stats["by_strength"].get(level, 0) + 1
        return stats
