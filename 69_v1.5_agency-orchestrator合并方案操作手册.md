## 二、架构设计

### 两层级联

| 层级 | 技能 | 职责 | 面向 |
|------|------|------|------|
| L1 | `agency_orchestrator` | MCP Server 7 工具直接封装 | 精细控制 |
| L2 | `auto_developer` | 一键开发（develop/compose/run） | 开箱即用 |

L2 内部调用 L1，不重复造轮子。

### 工具分布

**L1 agency_orchestrator（7 个工具）**

| 工具 | 执行方式 | 说明 |
|------|---------|------|
| list_roles | 本地读文件 | 直接读 external-skills/roles/ |
| get_role | 本地读文件 | 直接读 .md 文件 |
| list_workflows | 本地读文件 | 直接读 workflows/ 目录 |
| run_workflow | MCP 代理 → agency-orchestrator | 需要 AI 执行 |
| validate_workflow | MCP 代理 → agency-orchestrator | 服务端完整校验逻辑 |
| plan_workflow | MCP 代理 → agency-orchestrator | 服务端 DAG 构建逻辑 |
| compose_workflow | MCP 代理 → agency-orchestrator | 需要 AI 生成 YAML |

**L2 auto_developer（6 个工具）**

| 工具 | 执行方式 | 说明 |
|------|---------|------|
| develop | MCP 代理 → L1 run_workflow | 一键开发，跑 auto_develop.yaml |
| compose | MCP 代理 → L1 compose_workflow | 生成工作流 |
| run | MCP 代理 → L1 run_workflow | 运行指定工作流 |
| plan | MCP 代理 → L1 plan_workflow | 查看 DAG |
| list_roles | 代理 → L1 list_roles | 本地读文件 |
| list_workflows | 代理 → L1 list_workflows | 本地读文件 |

## 五、Commit 2：L1 技能 agency_orchestrator

### 4.1 skills/agency_orchestrator/implementation.py

```python
#!/usr/bin/env python3
"""
Agency Orchestrator 技能 - 多智能体 YAML 工作流引擎

功能：
    1. 执行/校验/规划/生成 YAML 工作流
    2. 列出 174 个 AI 角色，获取角色定义
    3. 列出 32 个内置工作流模板

L1 技能：MCP Server 7 工具的直接封装。
本地只读工具（list_roles/get_role/list_workflows）直接读文件；
需要 AI 执行的工具（run/validate/plan/compose）走 MCP 代理。

Author: Plector
Version: 1.1.0
"""

import asyncio
import logging
import re
from pathlib import Path
from typing import Any

from core.event_bus import get_event_bus

logger = logging.getLogger(__name__)

SKILL_DIR = Path(__file__).parent
PROJECT_DIR = SKILL_DIR.parent.parent
ROLES_DIR = PROJECT_DIR / "external-skills" / "roles"
WORKFLOWS_DIR = PROJECT_DIR / "servers" / "agency-orchestrator" / "workflows"

MCP_SERVER = "agency-orchestrator"


class SkillHandler:
    """Agency Orchestrator 技能处理器（L1 — MCP Server 封装）"""

    def __init__(self):
        self.name = "agency_orchestrator"

    # ─── 本地只读工具 ───

    def _list_roles_sync(self, category: str | None) -> dict[str, Any]:
        """同步：扫描角色目录"""
        if not ROLES_DIR.exists():
            return {"success": False, "data": None, "error": f"角色目录不存在: {ROLES_DIR}"}

        result = {}
        for cat_dir in sorted(ROLES_DIR.iterdir()):
            if not cat_dir.is_dir():
                continue
            if category and cat_dir.name != category:
                continue
            result[cat_dir.name] = self._scan_role_files(cat_dir)

        return {
            "success": True,
            "data": {
                "categories": list(result.keys()),
                "roles": result,
                "total": sum(len(v) for v in result.values()),
            },
            "error": None,
        }

    @staticmethod
    def _scan_role_files(cat_dir: Path) -> list[str]:
        """扫描分类目录下的角色文件名"""
        return [
            f.stem.replace(f"{cat_dir.name}-", "")
            for f in sorted(cat_dir.iterdir())
            if f.suffix == ".md"
        ]

    async def list_roles(self, category: str | None = None, **kwargs: Any) -> dict[str, Any]:
        """列出 AI 角色，按 18 个分类"""
        _ = kwargs
        try:
            loop = asyncio.get_event_loop()
            result = await loop.run_in_executor(None, self._list_roles_sync, category)

            bus = get_event_bus()
            await bus.publish(
                "roles.listed",
                {"category": category, "total": (result.get("data") or {}).get("total", 0)},
                source=self.name,
            )
            return result
        except Exception as e:
            logger.error(f"列出角色失败: {e}", exc_info=True)
            return {"success": False, "data": None, "error": str(e)}

    def _get_role_sync(self, category: str, name: str) -> dict[str, Any]:
        """同步：读取角色文件"""
        role_file = self._find_role_file(category, name)
        if not role_file:
            return {"success": False, "data": None, "error": f"角色不存在: {category}/{name}"}

        content = role_file.read_text(encoding="utf-8")
        fm_name, fm_desc = self._parse_frontmatter(content)

        return {
            "success": True,
            "data": {
                "category": category,
                "name": fm_name or name,
                "description": fm_desc,
                "content": content,
            },
            "error": None,
        }

    @staticmethod
    def _find_role_file(category: str, name: str) -> Path | None:
        """查找角色文件，尝试带前缀和不带前缀两种文件名"""
        for candidate in [
            ROLES_DIR / category / f"{name}.md",
            ROLES_DIR / category / f"{category}-{name}.md",
        ]:
            if candidate.exists():
                return candidate
        return None

    @staticmethod
    def _parse_frontmatter(content: str) -> tuple[str, str]:
        """解析 YAML frontmatter，返回 (name, description)"""
        fm_match = re.match(r"^---\n(.*?)\n---", content, re.DOTALL)
        if not fm_match:
            return ("", "")
        try:
            import yaml
            fm = yaml.safe_load(fm_match.group(1))
            return (fm.get("name", ""), fm.get("description", ""))
        except Exception:
            return ("", "")

    async def get_role(self, category: str, name: str, **kwargs: Any) -> dict[str, Any]:
        """获取指定角色的完整定义"""
        _ = kwargs
        try:
            loop = asyncio.get_event_loop()
            result = await loop.run_in_executor(None, self._get_role_sync, category, name)

            bus = get_event_bus()
            await bus.publish(
                "role.retrieved",
                {"category": category, "name": name},
                source=self.name,
            )
            return result
        except Exception as e:
            logger.error(f"获取角色失败: {e}", exc_info=True)
            return {"success": False, "data": None, "error": str(e)}

    def _list_workflows_sync(self) -> dict[str, Any]:
        """同步：扫描工作流目录"""
        if not WORKFLOWS_DIR.exists():
            return {"success": False, "data": None, "error": f"工作流目录不存在: {WORKFLOWS_DIR}"}

        templates = self._scan_workflow_files()
        return {
            "success": True,
            "data": {"count": len(templates), "templates": templates},
            "error": None,
        }

    @staticmethod
    def _scan_workflow_files() -> list[dict]:
        """递归扫描工作流目录，提取 name 和 description"""
        import yaml as _yaml
        templates = []
        for f in sorted(WORKFLOWS_DIR.rglob("*.yaml")):
            rel = f.relative_to(WORKFLOWS_DIR)
            try:
                doc = _yaml.safe_load(f.read_text(encoding="utf-8"))
                templates.append({
                    "name": str(rel).replace("\\", "/"),
                    "desc": doc.get("description", doc.get("name", "")),
                })
            except Exception:
                templates.append({"name": str(rel).replace("\\", "/"), "desc": "(解析失败)"})
        return templates

    async def list_workflows(self, **kwargs: Any) -> dict[str, Any]:
        """列出所有可用的工作流模板"""
        _ = kwargs
        try:
            loop = asyncio.get_event_loop()
            result = await loop.run_in_executor(None, self._list_workflows_sync)

            bus = get_event_bus()
            data = result.get("data") or {}
            await bus.publish(
                "workflows.listed",
                {"count": data.get("count", 0)},
                source=self.name,
            )
            return result
        except Exception as e:
            logger.error(f"列出工作流失败: {e}", exc_info=True)
            return {"success": False, "data": None, "error": str(e)}

    # ─── MCP 代理工具 ───

    async def run_workflow(
        self,
        path: str,
        inputs: dict | None = None,
        provider: str = "claude-code",
        model: str | None = None,
        resume: str | None = None,
        from_step: str | None = None,
        **kwargs: Any,
    ) -> dict[str, Any]:
        """执行 YAML 工作流（由 MCP Server 执行）"""
        _ = kwargs
        args: dict[str, Any] = {
            "path": path,
            "inputs": inputs or {},
            "provider": provider,
        }
        if model:
            args["model"] = model
        if resume:
            args["resume"] = resume
        if from_step:
            args["from_step"] = from_step

        bus = get_event_bus()
        await bus.publish(
            "workflow.executed",
            {"path": path, "provider": provider},
            source=self.name,
        )

        return {"_mcp_call": MCP_SERVER, "tool": "run_workflow", "args": args}

    async def validate_workflow(self, path: str, **kwargs: Any) -> dict[str, Any]:
        """校验 YAML 工作流文件（由 MCP Server 执行）"""
        _ = kwargs
        return {
            "_mcp_call": MCP_SERVER,
            "tool": "validate_workflow",
            "args": {"path": path},
        }

    async def plan_workflow(self, path: str, **kwargs: Any) -> dict[str, Any]:
        """显示 DAG 执行计划（由 MCP Server 执行）"""
        _ = kwargs
        return {
            "_mcp_call": MCP_SERVER,
            "tool": "plan_workflow",
            "args": {"path": path},
        }

    async def compose_workflow(
        self,
        description: str,
        provider: str = "claude-code",
        model: str | None = None,
        **kwargs: Any,
    ) -> dict[str, Any]:
        """用自然语言生成 YAML 工作流（由 MCP Server 执行）"""
        _ = kwargs
        args: dict[str, Any] = {"description": description, "provider": provider}
        if model:
            args["model"] = model

        bus = get_event_bus()
        await bus.publish(
            "workflow.composed",
            {"description": description[:100], "provider": provider},
            source=self.name,
        )

        return {"_mcp_call": MCP_SERVER, "tool": "compose_workflow", "args": args}
```

```python
#!/usr/bin/env python3
"""
自动开发技能 - 从需求到代码的全自动开发流水线

功能：
    1. 一键开发：输入需求，自动执行完整流水线
    2. 生成工作流：用自然语言生成 YAML
    3. 运行工作流：执行指定工作流
    4. 查看执行计划：显示 DAG

L2 技能：调用 agency_orchestrator (L1) 的 MCP 工具，提供高层封装。

Author: Plector
Version: 1.0.0
"""

import asyncio
import logging
from pathlib import Path
from typing import Any

from core.event_bus import get_event_bus

logger = logging.getLogger(__name__)

SKILL_DIR = Path(__file__).parent
PROJECT_DIR = SKILL_DIR.parent.parent
DEFAULT_WORKFLOW = PROJECT_DIR / "workflows" / "auto_develop.yaml"
AO_OUTPUT_DIR = PROJECT_DIR / "ao-output"

MCP_SERVER = "agency-orchestrator"


class SkillHandler:
    """自动开发技能处理器（L2 — 一键开发封装）"""

    def __init__(self):
        self.name = "auto_developer"

    async def develop(
        self,
        requirement: str,
        project_dir: str = ".",
        provider: str = "claude-code",
        **kwargs: Any,
    ) -> dict[str, Any]:
        """
        一键自动开发：输入需求，自动执行完整开发流水线。

        流程：产品经理→架构师+安全工程师(并行)→高级开发者→代码审查员→汇总
        """
        _ = kwargs

        if not DEFAULT_WORKFLOW.exists():
            return {
                "success": False,
                "data": None,
                "error": f"默认工作流不存在: {DEFAULT_WORKFLOW}",
            }

        bus = get_event_bus()
        await bus.publish(
            "auto_develop.started",
            {"requirement": requirement[:100], "provider": provider},
            source=self.name,
        )

        return {
            "_mcp_call": MCP_SERVER,
            "tool": "run_workflow",
            "args": {
                "path": str(DEFAULT_WORKFLOW),
                "inputs": {
                    "requirement": requirement,
                    "project_dir": project_dir,
                },
                "provider": provider,
            },
        }

    async def compose(
        self,
        description: str,
        provider: str = "claude-code",
        **kwargs: Any,
    ) -> dict[str, Any]:
        """根据描述自动生成工作流 YAML"""
        _ = kwargs

        bus = get_event_bus()
        await bus.publish(
            "workflow.composed",
            {"description": description[:100], "source": "auto_developer"},
            source=self.name,
        )

        return {
            "_mcp_call": MCP_SERVER,
            "tool": "compose_workflow",
            "args": {"description": description, "provider": provider},
        }

    async def run(
        self,
        workflow: str,
        inputs: dict | None = None,
        provider: str = "claude-code",
        **kwargs: Any,
    ) -> dict[str, Any]:
        """运行指定的工作流"""
        _ = kwargs

        bus = get_event_bus()
        await bus.publish(
            "workflow.executed",
            {"path": workflow, "source": "auto_developer"},
            source=self.name,
        )

        return {
            "_mcp_call": MCP_SERVER,
            "tool": "run_workflow",
            "args": {
                "path": workflow,
                "inputs": inputs or {},
                "provider": provider,
            },
        }

    async def plan(self, workflow: str, **kwargs: Any) -> dict[str, Any]:
        """查看工作流的 DAG 执行计划"""
        _ = kwargs
        return {
            "_mcp_call": MCP_SERVER,
            "tool": "plan_workflow",
            "args": {"path": workflow},
        }

    async def list_roles(self, category: str | None = None, **kwargs: Any) -> dict[str, Any]:
        """列出 AI 角色（代理 L1）"""
        from skills.agency_orchestrator.implementation import SkillHandler as L1
        return await L1().list_roles(category)

    async def list_workflows(self, **kwargs: Any) -> dict[str, Any]:
        """列出工作流模板（代理 L1）"""
        from skills.agency_orchestrator.implementation import SkillHandler as L1
        return await L1().list_workflows()

    def _read_latest_summary_sync(self) -> str | None:
        """同步：读取最新的 ao-output 摘要"""
        if not AO_OUTPUT_DIR.exists():
            return None

        dirs = sorted(
            [d for d in AO_OUTPUT_DIR.iterdir() if d.is_dir()],
            key=lambda d: d.stat().st_mtime,
            reverse=True,
        )

        if not dirs:
            return None

        summary_file = dirs[0] / "summary.md"
        return summary_file.read_text(encoding="utf-8") if summary_file.exists() else None

    async def read_summary(self, **kwargs: Any) -> dict[str, Any]:
        """读取最新工作流执行的摘要"""
        _ = kwargs
        try:
            loop = asyncio.get_event_loop()
            summary = await loop.run_in_executor(None, self._read_latest_summary_sync)

            if summary:
                return {"success": True, "data": {"summary": summary}, "error": None}
            return {"success": False, "data": None, "error": "没有找到执行结果"}
        except Exception as e:
            logger.error(f"读取摘要失败: {e}", exc_info=True)
            return {"success": False, "data": None, "error": str(e)}
```
