<script>
let ws = null;
let timer = null;

function connect() {
  const proto = location.protocol === 'https:' ? 'wss:' : 'ws:';
  ws = new WebSocket(proto + '//' + location.host + '/ws');
  ws.onopen = () => {
    document.getElementById('ws-dot').className = 'dot';
    document.getElementById('ws-status').textContent = '已连接';
    if (timer) { clearInterval(timer); timer = null; }
  };
  ws.onmessage = (e) => {
    const msg = JSON.parse(e.data);
    if (msg.type === 'thinking') addMsg('assistant', msg.content, 'thinking');
    else if (msg.type === 'response') { removeThinking(); addMsg('assistant', msg.content); }
    else if (msg.type === 'error') { removeThinking(); addMsg('error', msg.content, 'error'); }
    document.getElementById('send-btn').disabled = false;
  };
  ws.onclose = () => {
    document.getElementById('ws-dot').className = 'dot off';
    document.getElementById('ws-status').textContent = '已断开';
    if (!timer) timer = setInterval(connect, 3000);
  };
}

function sendMsg() {
  const input = document.getElementById('input');
  const content = input.value.trim();
  if (!content || !ws || ws.readyState !== 1) return;
  addMsg('user', content);
  ws.send(JSON.stringify({ content }));
  input.value = '';
  document.getElementById('send-btn').disabled = true;
}

function addMsg(role, content, type) {
  const box = document.getElementById('messages');
  const div = document.createElement('div');
  div.className = 'message ' + (type || role);
  const label = role === 'user' ? '你' : role === 'error' ? '错误' : 'Plector';
  div.innerHTML = '<div class="role">' + label + '</div><div class="content">' + esc(content) + '</div>';
  box.appendChild(div);
  box.scrollTop = box.scrollHeight;
}

function removeThinking() {
  const t = document.querySelector('.message.thinking');
  if (t) t.remove();
}

function esc(text) {
  const d = document.createElement('div');
  d.textContent = text;
  return d.innerHTML;
}

function togglePanel(header) {
  header.nextElementSibling.classList.toggle('hide');
}

async function fetchApi(url) {
  try { return await (await fetch(url)).json(); } catch { return null; }
}

function barColor(v) { return v < 60 ? 'var(--green)' : v < 80 ? 'var(--yellow)' : 'var(--red)'; }

async function updateHealth() {
  const d = await fetchApi('/api/health');
  if (!d) return;
  ['cpu', 'mem', 'disk'].forEach(k => {
    document.getElementById(k + '-value').textContent = d[k === 'mem' ? 'memory' : k] + '%';
    const bar = document.getElementById(k + '-bar');
    bar.style.width = d[k === 'mem' ? 'memory' : k] + '%';
    bar.style.background = barColor(d[k === 'mem' ? 'memory' : k]);
  });
}

async function updateSkills() {
  const d = await fetchApi('/api/skills');
  if (!d) return;
  document.getElementById('skills-list').innerHTML = d.skills.map(s =>
    '<li><span>' + s.name + '</span><span class="badge">' + s.tools.length + ' tools</span></li>'
  ).join('');
}

async function updateTools() {
  const d = await fetchApi('/api/tools');
  if (!d) return;
  document.getElementById('tools-list').innerHTML = [
    ...d.local.map(t => '<li><span>' + t.name + '</span><span class="badge">local</span></li>'),
    ...d.mcp.map(t => '<li><span>' + t.name + '</span><span class="badge">mcp</span></li>'),
  ].join('');
  document.getElementById('tools-count').textContent = d.total + ' tools';
}

async function updateConfig() {
  const d = await fetchApi('/api/config');
  if (d) document.getElementById('llm-provider').textContent = d.llm_provider.toUpperCase();
}

async function updateEvents() {
  const d = await fetchApi('/api/events');
  if (!d) return;
  document.getElementById('events-list').innerHTML = d.events.slice(-20).reverse().map(e => {
    const t = e.time.split('T')[1].split('.')[0];
    return '<div class="event-item"><span class="time">' + t + '</span> <span class="type">' + e.type + '</span></div>';
  }).join('');
}

document.getElementById('input').addEventListener('keydown', e => {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendMsg(); }
});

connect();
updateHealth(); updateSkills(); updateTools(); updateConfig();
setInterval(updateHealth, 5000);
setInterval(updateEvents, 3000);
</script>
